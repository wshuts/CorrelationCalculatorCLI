function getDate(dateString) {

      var day, month, year;
      var delimiter = "";
      var index = 0;
      if ((index = dateString.indexOf("/")) >= 1) { delimiter = "/";
      } else if ((index = dateString.indexOf("-")) >= 1) { delimiter = "-";
      }
      if (delimiter != "") {
         month = dateString.substring(0, index);
         day = dateString.substring(index + 1, index = dateString.indexOf(delimiter, index+1));
         year = dateString.substring(index+1);
      } else {
                month = dateString.substring(0, 2);
                day = dateString.substring(2, 4);
                year = dateString.substring(4);
      }
      return new Date(year, month-1, day);
}

function isDatePriceHistory(dateString){
    var day, month, year;
      var delimiter = "";
      var index = 0;
      var length = dateString.length;
      if(length>10||length<8){
          return false;

      }
      if ((index = dateString.indexOf("/")) >= 1) { delimiter = "/";
      }else{
          return false;
      }

       month = dateString.substring(0, index);
       day = dateString.substring(index + 1, index = dateString.indexOf(delimiter, index+1));
       year = dateString.substring(index+1);
       var c=new Date(year,month-1,day);
       if(year.length!=4||c.getMonth()!=month-1||c.getDate()!=day||c.getFullYear()!=year){
            return false
       }
       return true;

}
